        def    is_dos33?
                # VTOC is at offset 0x11000
                # bytes 1/2/3 are a track number, sector number and DOS version number
                # see if these are reasonable values

                (@file_bytes[0x11001]<=34) && (@file_bytes[0x11002]<=15) && (@file_bytes[0x11003]==3)
        end
